```python
import re
from collections import Counter

def contar_frequencia_palavras(texto):
    # Transforma o texto em minúsculas
    texto = texto.lower()
    
    # Extrai palavras da string e contabiliza sua frequência
    word_counts = Counter(re.findall(r'\b\w+\b', texto))
    
    return word_counts

def exibir_palavras_mais_comuns(word_counts, n=10):
    print("Top 10 palavras mais comuns:")
    for word, count in word_counts.most_common(n):
        print(f"{word}: {count}")
```

```python
import re
from collections import Counter

def contar_frequencia_palavras(texto):
    # Transforma o texto em minúsculas
    texto = texto.lower()
    
    # Extrai palavras da string e contabiliza sua frequência
    word_counts = Counter(re.findall(r'\b\w+\b', texto))
    
    return word_counts

def exibir_palavras_mais_comuns(word_counts, n=10):
    print("Top 10 palavras mais comuns:")
    for word, count in word_counts.most_common(n):
        print(f"{word}: {count}")

# Chama a função para analisar o texto
texto = "Olá, mundo! Eu amo o mundo. Aprender Python é divertido! Olá novamente!"
palavras Mais Comuns = contar_frequencia_palavras(texto)
exibir_palavras_mais_comuns(palavrasMaisComuns, 10)

# Chama a função com um texto exemplo
texto_exemplo = "A linguagem de programação mais usada é Python. A Python é uma linguagem muito popular."
palavrasMaisComunsExemplo = contar_frequencia_palavras(texto_exemplo)
exibir_palavras_mais_comuns(palavrasMaisComunsExemplo, 10)

with open('texto.txt', 'r') as arquivo:
```
